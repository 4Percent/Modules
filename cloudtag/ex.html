Canvas版,可添加点击事件,4种默认图形,点一下试试?

			new CloudTagCanvas(document.getElementById('canvas'),{
				perspective : 1000,		//可空,景深,默认值1000
				r : 200,		//不可空,半径
				detectR : 300,		//可空,球心距离鼠标多远时开始做出反应
				centerX : 400,		//不可空,球心的X坐标
				centerY : 400,		//不可空,球心的Y坐标
				points : [		//球面上的标签
					{
						type : 'star',		//不可空,图形类型,此处为星星
						bg : '#000',		//不可空,初始背景
						hbg : '#fff',		//可空,鼠标hover时的背景
						innerR : 20,		//不可空,星星的内半径
						outerR : 50,		//不可空,星星的外半径
						text : 'Star',		//可空,标签内的文字
						color : '#333',		//可空,文字的颜色
						handler : function		//可空,鼠标点击时的处理函数,函数内的this为当前的json对象
					},
					{
						type : 'rect',		//矩形
						bg : '#000',
						hbg :'#fff',
						w : 100,		//不可空,矩形的宽
						h : 80,		//不可空,矩形的高
						text : 'Rect',
						color : '#fff',
						handler : function
					},
					{
						type : 'hexagon' 或者 'round',		//正六边形或圆
						bg : '#000',
						hbg : '#fff',
						r : 100,		//不可空,六边形或圆的半径
						text : 'Round',
						color : '#333',
						handler : function
					},
					{
						type : 'image',		//图片
						src : 'unclecat.jpg',		//不可空,图片的src
						w : 100,
						h : 100,
						text : 'Uncle Cat',
						color : '#333',
						handler : function
					}
				]
			});
